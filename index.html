    ax.get_yaxis().tick_left()

def plot_reaction_bar(ratio, ylim, fig_name):
    ratios = [ratio.loc[reaction].values for reaction in ['LIKE', 'LOVE', 'HAHA', 'WOW', 'SAD', 'ANGRY']]
    print(ratios)
    plt.bar(range(6), [r * 100 for r in ratios], color='k')
    ax = plt.gca()
    ax.set_xticklabels(['', 'Like', 'Love', 'Haha', 'Wow', 'Sad', 'Angry'])
    ax.yaxis.set_tick_params(labelsize=30)
    ax.xaxis.set_tick_params(labelsize=30)
    box_off()
    ax.set_ylim(ylim)
    ax.set_title(fig_name, fontsize=30, fontweight='bold')
    ax.set_ylabel('Reações + comentários (%)', fontsize=30, fontweight='bold')
